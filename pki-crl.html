var pki = new GostPKIX(provider.value, 'PEM');
pki.importCertificate(certAndKeyCA.textContent, 'Pushkin').then(function() {
    
    // Import root CA private key
    return pki.importKey(certAndKeyCA.textContent, 'Pushkin');
}).then(function() {

    // Create list of certificates to add
    revokedCertificates = [];
    var today = new Date();
    today.setHours(0, 0, 0, 0);
    var list = revocationList.textContent.split('\n');
    for (var i = 0; i < list.length; i++) {
        if (list[i]) {
            s = list[i].replace(/[^a-zA-Z0-9\:]/g, '').split(':');
            revokedCertificates.push({
                userCertificate: s[0],
                revocationDate: today,
                crlEntryExtensions: {
                    invalidityDate: today,
                    cRLReason: s[1]
                }
            });
        }
    }

    // Create new or update current CRL
    return pki.updateCRL({
        thisUpdate: today,
        revokedCertificates: revokedCertificates,
        crlExtensions: {cRLNumber: 1}
        }, 'Pushkin');
}).then(function(data) {

    // Output ready CRL
    CRL.textContent = data;

    // Reimport issured CRL (for verification import procedure)
    return pki.importCRL(data, 'Pushkin');
}).catch(function(reason) {
    alert(reason.message);
});
