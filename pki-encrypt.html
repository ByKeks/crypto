                    Encrypt data method
                    
var pki = new GostPKIX(provider.value, 'PEM');
var content = gostCoding.Chars.decode(message.textContent, 'utf-8');
var mode = encryptionType.value;
var promise;
switch(mode) {
    case 'KEYAGREE':
    case 'KEYTRANS':
        // Use certificate
        pki.keyStore.setCertificate('Pushkin', recipient.textContent);
        promise = pki.encryptData(content, mode, 'Pushkin');
        break;
    case 'KEK':
    case 'KEYMAN':
        // Use key encryption key
        pki.keyStore.setKey('Pushkin', gostCoding.Hex.decode(kek.value));
        promise = pki.encryptData(content, mode, 'Pushkin');
        break;
    case 'PBKEK':
    case 'PBES':
        promise = pki.encryptData(content, mode, password.value);
        break;
}
promise.then(function(data) {

    // Output encrypted data
    encryptedData.textContent = data;

}).catch(function(reason) {
    alert(reason.message);
});

var pki = new GostPKIX(provider.value, 'PEM');
var data = encryptedData.textContent;
var mode = encryptionType.value;
var promise;
switch(mode) {
    case 'KEYAGREE':
    case 'KEYTRANS':
        // Use certificate
        pki.keyStore.setKey('Pushkin', recipient.textContent); 
        pki.keyStore.setCertificate('Pushkin', recipient.textContent); 
        promise = pki.extractData(data, 'Pushkin');
        break;
    case 'KEK':
    case 'KEYMAN':
        // Use key encryption key
        pki.keyStore.setKey('Pushkin', gostCoding.Hex.decode(kek.value));
        promise = pki.extractData(data, 'Pushkin');
        break;
    case 'PBES':
    case 'PBKEK':
        promise = pki.extractData(data, password.value);
        break;
}
promise.then(function(data) {

    // Decrypted result
    decryptedData.textContent = gostCoding.Chars.encode(data.content, 'utf-8');

}).catch(function(reason) {
    alert(reason.message);
});
