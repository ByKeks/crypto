// Validity period
var notBefore = new Date();
notBefore.setHours(0, 0, 0, 0);
var notAfter = new Date(notBefore);
notAfter.setDate(notAfter.getDate() + parseInt(days.value || '365'));

// Create CA certificate with parameters from this page
var pki = new GostPKIX(provider.value, 'PEM');
pki.createCertificate({
    validity: {
        notBefore: notBefore,
        notAfter: notAfter
    },
    subject: {
        countryName: countryName.value,
        stateOrProvinceName: stateOrProvinceName.value,
        organizationName: organizationName.value,
        organizationalUnitName: organizationalUnitName.value,
        title: title.value,
        commonName: commonName.value
    },
    extensions: {
        keyUsage: ['digitalSignature', 'nonRepudiation', 'keyEncipherment', 
            'dataEncipherment', 'keyAgreement', 'keyCertSign', 'cRLSign'],
        extKeyUsage: ['serverAuth', 'clientAuth', 'codeSigning', 'emailProtection']
    }
}, 'Pushkin').then(function(data) {

    // Output ready certificate
    certificate.textContent = data;

    // Output ready private key
    return pki.exportKey('Pushkin');
}).then(function(data) {

    certificate.textContent += '\r\n\r\n' + data;
}, function(reason) {
    alert(reason.message);
});
