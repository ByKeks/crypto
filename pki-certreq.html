// Create certificate request with parameters from this page
var pki = new GostPKIX(provider.value, 'PEM');
pki.createRequest({
    subject: {
        countryName: countryName.value,
        stateOrProvinceName: stateOrProvinceName.value,
        organizationalUnitName: organizationalUnitName.value,
        organizationName: organizationName.value,
        title: title.value,
        commonName: commonName.value
    }
}, 'Onegin').then(function(data) {

    // Output ready certificate
    request.textContent = data;

    // Output ready private key
    return pki.exportKey('Onegin');
}).then(function(data) {

    privateKey.textContent = data;

// Import root CA certificate
var pki = new GostPKIX(provider.value, 'PEM');
pki.importCertificate(certAndKeyCA.textContent, 'Pushkin').then(function() {
    
    // Import root CA private key
    return pki.importKey(certAndKeyCA.textContent, 'Pushkin');
}).then(function() {

    // Import certificate request
    return pki.importRequest(request.textContent, 'Onegin');
}).then(function() {

    // Validity period
    var validity = {};
    validity.notBefore = new Date();
    validity.notBefore.setHours(0, 0, 0, 0);
    validity.notAfter = new Date(validity.notBefore);
    validity.notAfter.setDate(validity.notAfter.getDate() + parseInt(days.value || '365'));

    // Issue certificate
    return pki.issueCertificate({validity: validity}, 'Onegin', 'Pushkin');
}).then(function(data) {

    // Output ready certificate
    certificate.textContent = data + '\r\n\r\n' + privateKey.textContent;

}).catch(function(reason) {
    alert(reason.message);
});
